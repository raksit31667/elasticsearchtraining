##### Custom Analyzer #####
DELETE my_index

PUT my_index
{
  "settings": {
    "analysis": {
      "analyzer": {
        "std_folded": {
          "type": "custom",
          "tokenizer": "standard",
          "filter": [
            "lowercase",
            "asciifolding"
          ]
        }
      }
    }
  },
  "mappings": {
    "properties": {
      "my_text": {
        "type": "text",
        "analyzer": "std_folded"
      }
    }
  }
}

GET my_index

GET my_index/_analyze
{
  "analyzer": "std_folded",
  "text":     "Is this déjà vu?"
}

GET my_index/_analyze
{
  "field": "my_text",
  "text":  "Is this déjà vu?"
}
